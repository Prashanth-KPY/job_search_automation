# Bing Search
BING_SEARCH_ENDPOINT=https://api.bing.microsoft.com/v7.0/search

SMTP_PASS=your_app_password   # use app password with 2FA enabled
EMAIL_FROM=your@gmail.com
EMAIL_TO=recipient@example.com

# Optional
